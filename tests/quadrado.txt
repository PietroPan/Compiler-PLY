main {
    int r
    int init
    int inp
    int i

    prints("Insira 4 números:\n")
    read(init)
    inp=init

    for (i=0;i<3;i++){
        read(inp)
        r+=(inp==init)
    }
    if (r==3){
        prints("Números introduzidos podem ser lados de um quadrado\n")
    } else {
        prints("Números introduzidos não podem ser lados de um quadrado\n")
    }
}
